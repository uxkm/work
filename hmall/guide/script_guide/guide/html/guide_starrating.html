StarRating


html문서 내 호출 및 기본 구조

	
	<!-- 기본 -->
	<span class="star_rating_box int" data-modules-starRating>
		<a href="#" class="star ui-active">1</a>
		<a href="#" class="star ui-active fix">2</a>
		<a href="#" class="star">3</a>
		<a href="#" class="star">4</a>
		<a href="#" class="star">5</a>
		<span class="num">2</span>
		<span class="maximum"><i>/</i> 5</span>
	</span>

script에서 호출

	
	// 기본 옵션없이 호출
	$(Element).starRating();


	// 커스텀 옵션 사용
	$(Element).moreview({
		"score":0.5
	});
