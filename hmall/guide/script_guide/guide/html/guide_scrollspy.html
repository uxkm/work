scrollspy


html문서 내 호출 및 기본 구조

	
	<!-- 기본 -->
	<div class="scrollspy-tab" data-modules-scrollspy="padding:102;">
		<div class="inner">
			<!-- Nav tabs -->
			<ul class="nav-tabs ui-spynav" role="tablist">
				<li role="presentation" class="ui-active"><a href="#viewpage1"><span>콘텐츠1</span></a></li>
				<li role="presentation"><a href="#viewpage2"><span>콘텐츠2</span></a></li>
				<li role="presentation"><a href="#viewpage3"><span>콘텐츠3</span></a></li>
				<li role="presentation"><a href="#viewpage4"><span>콘텐츠4</span></a></li>
			</ul>
		</div>
	</div>
	<div class="pages page1" id="viewpage1">1</div>
	<div class="pages page2" id="viewpage2">2</div>
	<div class="pages page3" id="viewpage3">3</div>
	<div class="pages page4" id="viewpage4">4</div>

script에서 호출

	
	// 기본 옵션없이 호출
	$(Element).scrollspy();


	// 커스텀 옵션 사용
	$(Element).scrollspy({
		'padding':102
	});

최소한의 CSS

	
	.scrollspy-tab.fixed .inner{position: fixed;top: 0;left: 0;right: 0;}
	.nav-tabs{display: flex;}
	.nav-tabs li{flex:1;}
	.pages{height: 800px;margin-top: 100px;}

options

Name Type default description
durration int 200 스크롤 애니메이션 속도
margin int 0 네비게이션이 브라우저 상단으로 부터 {margin}값 미만으로 접근시 상단에 고정된다.
padding int 0 컨텐츠가 브라우저 상단으로부터 {padding} 값 미만으로 접근시 해당 네비게이션이 활성화 된다.
