Custom selectbox


html문서 내 호출 및 기본 구조

	
	<!-- 기본 -->
	<div class="custom-selectbox" data-modules-selectbox>
		<select>
			<option value="aa1">내용1</option>
			<option value="aa2">내용2</option>
			<option value="aa3">내용3</option>
		</select>
	</div>

script에서 호출

	
	$(Element).selectbox();

최소한의 CSS

	
.custom-selectbox{
	position: relative;
}
.custom-selectbox select{
	width: 0;
	font-size: 0;
	opacity: 0;
	height: 0;
	position: absolute;
	top: -9999em;
	left: -9999em;
}
.custom-selectbox .ui-selectbox{
	display: none;
}
