script에서 호출

	
	// 기본 옵션없이 호출
	$(Element).sticky();


	// 커스텀 옵션 사용
	$(Element).sticky({
		padding:0,
		relativeTarget : '#wrap'
	});

최소한의 CSS

	
.ui-tab{
	white-space: nowrap;
}
.tab-content > .tab-pane.ui-active{
	display: block;
}
.tab-content > .tab-pane {
	display: none;
}
.ui-tab li{display: inline-block;}

.scroll-tab{
	position: relative;
	width: 464px;
}
.scroll-tab .inner-scroll{
	overflow:hidden;
}
.scroll-tab .ui-tab{
	margin: 0 auto;
}
.scroll-tab.next-disabled .next-tab,
.scroll-tab.prev-disabled .prev-tab,
.scroll-tab .scrolltab-btns{
	display: none;
}
.scroll-tab.ui-active .scrolltab-btns{
	display: block;
}
