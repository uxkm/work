Tooltip


html문서 내 호출 및 기본 구조

	
	<!-- default(click) -->
	<div class="ui-tooltip">
		<span class="nav" data-modules-tooltip="href:#tooltip1;">tooltip click type</span>
		<div id="tooltip1" class="tooltip-conts">tooltip contents</div>
	</div>

	<!-- hover -->
	<div class="ui-tooltip">
		<a href="#tooltip2" class="nav" data-modules-tooltip="type:hover;">tooltip hover type</a>
		<div id="tooltip2" class="tooltip-conts">tooltip contents</div>
	</div>

	<!-- toggle -->
	<div class="ui-tooltip">
		<a href="#tooltip3" class="nav" data-modules-tooltip="toggle:true;">tooltip toggle type</a>
		<div id="tooltip3" class="tooltip-conts">tooltip contents</div>
	</div>

script에서 호출

	
	// 기본 옵션없이 호출
	$(Element).tooltip();


	// 커스텀 옵션 사용
	$(Element).tooltip({
		"toggle":true
	});

최소한의 CSS

	
.ui-tooltip{position: relative;text-align: left;display: inline-block;margin: 0 20px;}
.ui-tooltip .nav{border:1px solid #000;display: inline-block;padding: 20px;}
.ui-tooltip .tooltip-conts{display: none;width: 300px;text-align: center;font-size: 80px;border: 1px solid #000;position: absolute;top: 100%;left: 0;background: #fff;z-index: 999;}
.ui-tooltip .tooltip-conts button{font-size: 16px;line-height: 100%;position: absolute;top: 0;right: 0;}
.ui-tooltip .tooltip-conts.ui-active{display: block;}

options

Name Type default description
type string click click, hover 툴팁 오픈타입 지정
href selector null 기본은 a링크의 해시값으로 지정 하나, a링크가 아닌 태그 사용시 옵션값으로 지정 가능
closeBtn Boolean true 닫기버튼 삽입 여부
toggle Boolean false 튤팁 토글 여부 선택
