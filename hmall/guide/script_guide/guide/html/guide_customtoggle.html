Customtoggle


html문서 내 호출 및 기본 구조

	
	<!-- 기본(자신이 토글되는 경우) -->
	<button type="button" data-modules-customtoggle>커스텀 토글</button>

	
	<!-- 커스텀 옵션 사용(부모요소 토글) -->
	<div class="parent-wrap">
		<div class="toggle-body">내용.....</div>
		<button type="button" data-modules-customtoggle="parent:.parent-wrap">커스텀 토글</button>
	</div>
	
	
	<!-- 커스텀 옵션 사용(부모요소 토글, 텍스트 + 닫기버튼) -->
	<div class="parent-wrap">
		<div class="toggle-body">내용.....</div>
		<button type="button" data-modules-customtoggle="parent:.parent-wrap">
			커스텀 토글
			<span data-text>더보기</span>
		</button>
		<span data-close-btn>닫기</span>	
	</div>

script에서 호출

	
	// 기본 옵션없이 호출
	$(Element).customtoggle();


	// 커스텀 옵션 사용
	$(Element).customtoggle({
		'parent':'.parent-wrap'
	});

최소한의 CSS

	
[data-modules-customtoggle].ui-active{
	background: #000;
	color: #fff;
}
.parent-wrap.ui-active  .toggle-body{
	display: block;
}
.parent-wrap .toggle-body{
	display: none;
}

options

Name Type default description
parent selector null 토글 시킬 셀렉터, 기본은 자신입니다.
activeClass string ui-active 토글 시킬 클래스 이름
openText String 더보기 클래스 토글과 함께 변경할 문구가 있을 시, activeClass 클래스 요소가 활성화 되면 적용됩니다 !
closeText String 닫기 클래스 토글과 함께 변경할 문구가 있을 시, activeClass 클래스 요소가 제거되면 적용됩니다, !

Method

Name description
$(element).customtoggle('toggle') 토글 메서드
$(element).customtoggle('show') 토글 활성화
$(element).customtoggle('hide') 토글 비활성화

Event

Name description
$(element).on('toggle.customtoggle') 토글 이벤트
